# Aakha: assistive navigation stack.
# Pinned against a Python 3.12 venv on macOS arm64 (Apple Silicon).
#
#   python3.12 -m venv .venv
#   source .venv/bin/activate
#   pip install -r requirements.txt
#
# NOTE: OCR uses Apple's Vision framework via `ocrmac` (macOS-only, native to
# Apple Silicon). No system binary or model download needed. The backend always
# runs on the Mac, so a macOS-native OCR engine is fine.

numpy==2.5.1
pillow==12.3.0

# --- Tier 1: vision (detection + tracking) ---
opencv-python==5.0.0.93
ultralytics==8.4.91          # YOLO11n + ByteTrack (model.track(persist=True))
lap==0.5.13                  # linear-assignment solver ByteTrack needs for tracking
onnx==1.22.0                 # export YOLO11n -> ONNX
onnxslim==0.1.94             # graph-slim step Ultralytics runs during ONNX export
onnxruntime==1.27.0          # ONNX inference runtime
torch==2.13.0                # pulled by ultralytics; also used directly by Tier 2
torchvision==0.28.0

# --- audio out: TTS ---
pyttsx3==2.99                # offline TTS (NSSpeechSynthesizer on macOS)

# --- web layer: phone camera client + dashboard (Dev 3 / mobile-client) ---
fastapi==0.121.2             # WS endpoints (/camera, /status, /control) + pages
uvicorn[standard]==0.40.0    # ASGI server; [standard] pulls websockets + uvloop
